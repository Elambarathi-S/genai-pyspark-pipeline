# SALESANALYTICS CLASS METHODS
# ============================================================================

"""
1. CREATE_SPARK_SESSION()
   Purpose: Get or create configured Spark session
   Returns: SparkSession
   Configuration:
   - spark.executor.memory = 4g
   - spark.driver.memory = 4g
   - spark.serializer = KryoSerializer
   - spark.sql.adaptive.enabled = true
   - spark.master = local[4]
   
2. LOAD_PARQUET(path: str) → DataFrame
   Purpose: Load Parquet files into Spark DataFrame
   Validation: Checks file existence before loading
   Logging: Row count logged on successful load
   Errors: Raises FileNotFoundError if file missing
   
3. TOP_CUSTOMERS_BY_REVENUE(orders_df, products_df, n=10) → DataFrame
   Purpose: Find top N customers by total revenue
   Algorithm:
   - Join orders with products
   - Calculate line revenue (quantity × price)
   - Group by customer, aggregate revenue/orders/avg
   - Sort by revenue DESC, limit to N
   Columns Returned:
   - customer_id (INT)
   - total_revenue (DOUBLE, 2 decimals)
   - num_orders (LONG)
   - avg_order_value (DOUBLE, 2 decimals)
   
4. SALES_BY_CATEGORY(orders_df, products_df) → DataFrame
   Purpose: Aggregate sales metrics by product category
   Algorithm:
   - Join orders with products
   - Calculate line revenue
   - Group by category
   - Sum revenue and units, count transactions
   - Sort by revenue DESC
   Columns Returned:
   - category (STRING)
   - total_revenue (DOUBLE, 2 decimals)
   - total_units_sold (LONG)
   - num_transactions (LONG)
   - avg_transaction_value (DOUBLE, 2 decimals)
   
5. MONTHLY_TRENDS(orders_df, products_df) → DataFrame
   Purpose: Calculate month-over-month revenue trends with growth %
   Algorithm:
   - Join orders with products
   - Calculate revenue and extract month
   - Group by month, sum revenue
   - Apply Window function: lag(revenue) for previous month
   - Calculate: growth% = ((current - previous) / previous) × 100
   Window Functions Used:
   - lag("total_revenue").over(Window.orderBy("year_month"))
   - Enables YoY growth calculation
   Columns Returned:
   - year_month (STRING, YYYY-MM-DD)
   - total_revenue (DOUBLE, 2 decimals)
   - prev_revenue (DOUBLE, 2 decimals)
   - growth_percentage (DOUBLE, 2 decimals or NULL)
"""
